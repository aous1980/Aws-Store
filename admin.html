// إعدادات GitHub - ضع بياناتك هنا const GITHUB_TOKEN = 'ضع_التوكن_الخاص_بك_هنا'; const REPO_OWNER = 'اسم_حسابك'; const REPO_NAME = 'اسم_المستودع'; const FILE_PATH = 'data.json'; const API_URL = `https://api.github.com/repos/${REPO_OWNER}/${REPO_NAME}/contents/${FILE_PATH}`; let editIndex = -1; let currentSha = ""; // ضروري لتحديث ملفات GitHub // دالة لجلب البيانات من GitHub عند تحميل الصفحة async function loadFromGitHub() { try { const res = await fetch(API_URL); const fileData = await res.json(); currentSha = fileData.sha; const content = JSON.parse(atob(fileData.content)); // فك تشفير البيانات renderList(content); return content; } catch (e) { console.error("خطأ في جلب البيانات", e); return []; } } // دالة لحفظ البيانات إلى GitHub async function saveToGitHub(newData) { const updatedContent = btoa(JSON.stringify(newData, null, 2)); // تشفير البيانات const res = await fetch(API_URL, { method: 'PUT', headers: { 'Authorization': `token ${GITHUB_TOKEN}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ message: "تحديث قائمة التطبيقات", content: updatedContent, sha: currentSha }) }); if (res.ok) { alert("✅ تم التحديث بنجاح على السيرفر!"); location.reload(); // لإعادة جلب SHA الجديد } else { alert("❌ فشل التحديث، تأكد من الإعدادات"); } } // تعديل زر الحفظ ليستخدم GitHub document.getElementById('btnSave').onclick = async function() { const name = document.getElementById('appName').value; const url = document.getElementById('appUrl').value; const img = document.getElementById('appImg').value; const cat = document.getElementById('category').value; let data = await loadFromGitHub(); if (editIndex === -1) { data.push({ name, url, img, cat }); } else { data[editIndex] = { name, url, img, cat }; } await saveToGitHub(data); }; // تشغيل الجلب عند البداية window.onload = loadFromGitHub;
